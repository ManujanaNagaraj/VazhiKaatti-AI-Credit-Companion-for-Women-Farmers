# ============================================================================
# INSTALLATION & RUN INSTRUCTIONS
# ============================================================================
#
# STEP 1: INSTALL BACKEND DEPENDENCIES
# -------------------------------------
# cd backend
# pip install -r requirements.txt
#
# Alternative (if using virtual environment):
# python -m venv venv
# venv\Scripts\activate          # Windows
# source venv/bin/activate       # Mac/Linux
# pip install -r requirements.txt
#
#
# STEP 2: TRAIN THE ML MODEL
# ---------------------------
# cd backend
# python train_model.py
#
# This will:
# - Generate sample training data (if not exists)
# - Train the Random Forest credit scoring model
# - Save model files to models/credit_model.pkl and models/scaler.pkl
#
#
# STEP 3: START THE BACKEND SERVER
# ---------------------------------
# Option A - Using start script:
# python start_server.py
#
# Option B - Direct uvicorn command:
# uvicorn main:app --reload --host 0.0.0.0 --port 8000
#
# Option C - Using Python module:
# python -m uvicorn main:app --reload
#
# Backend will run at: http://localhost:8000
# API Documentation: http://localhost:8000/docs
#
#
# STEP 4: START THE FRONTEND (React)
# -----------------------------------
# Open a NEW terminal window, then:
# cd frontend
# npm install              # First time only
# npm start
#
# Frontend will run at: http://localhost:3000 or http://localhost:3001
#
#
# QUICK START (All-in-One)
# -------------------------
# Terminal 1 (Backend):
#   cd backend
#   pip install -r requirements.txt
#   python train_model.py
#   python -m uvicorn main:app --reload
#
# Terminal 2 (Frontend):
#   cd frontend
#   npm install
#   npm start
#
#
# TESTING THE API
# ---------------
# curl http://localhost:8000/
# curl http://localhost:8000/match-schemes?score=75
#
#
# TROUBLESHOOTING
# ---------------
# If port 8000 is busy:
#   uvicorn main:app --reload --port 8001
#
# If model files are missing:
#   python train_model.py
#
# If dependencies fail to install:
#   pip install --upgrade pip
#   pip install -r requirements.txt --no-cache-dir
#
# ============================================================================
